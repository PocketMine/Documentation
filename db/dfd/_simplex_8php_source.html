96  public function getNoise3D($x, $y, $z){
97  $x += $this->offsetX;
98  $y += $this->offsetY;
99  $z += $this->offsetZ;
100 
101  // Skew the input space to determine which simplex cell we're in
102  $s = ($x + $y + $z) * self::$F3; // Very nice and simple skew factor for 3D
103  $i = (int) floor($x + $s);
104  $j = (int) floor($y + $s);
105  $k = (int) floor($z + $s);
106  $t = ($i + $j + $k) * self::$G3;
107  $X0 = $i - $t; // Unskew the cell origin back to (x,y,z) space
108  $Y0 = $j - $t;
109  $Z0 = $k - $t;
110  $x0 = $x - $X0; // The x,y,z distances from the cell origin
111  $y0 = $y - $Y0;
112  $z0 = $z - $Z0;
113 
114  // For the 3D case, the simplex shape is a slightly irregular tetrahedron.
115 
116  // Determine which simplex we are in.
117  if($x0 >= $y0){
118  if($y0 >= $z0){
119  $i1 = 1;
120  $j1 = 0;
121  $k1 = 0;
122  $i2 = 1;
123  $j2 = 1;
124  $k2 = 0;
125  } // X Y Z order
126  elseif($x0 >= $z0){
127  $i1 = 1;
128  $j1 = 0;
129  $k1 = 0;
130  $i2 = 1;
131  $j2 = 0;
132  $k2 = 1;
133  } // X Z Y order
134  else{
135  $i1 = 0;
136  $j1 = 0;
137  $k1 = 1;
138  $i2 = 1;
139  $j2 = 0;
140  $k2 = 1;
141  }
142  // Z X Y order
143  }else{ // x0<y0
144  if($y0 < $z0){
145  $i1 = 0;
146  $j1 = 0;
147  $k1 = 1;
148  $i2 = 0;
149  $j2 = 1;
150  $k2 = 1;
151  } // Z Y X order
152  elseif($x0 < $z0){
153  $i1 = 0;
154  $j1 = 1;
155  $k1 = 0;
156  $i2 = 0;
157  $j2 = 1;
158  $k2 = 1;
159  } // Y Z X order
160  else{
161  $i1 = 0;
162  $j1 = 1;
163  $k1 = 0;
164  $i2 = 1;
165  $j2 = 1;
166  $k2 = 0;
167  }
168  // Y X Z order
169  }
170 
171  // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in (x,y,z),
172  // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in (x,y,z), and
173  // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in (x,y,z), where
174  // c = 1/6.
175  $x1 = $x0 - $i1 + self::$G3; // Offsets for second corner in (x,y,z) coords
176  $y1 = $y0 - $j1 + self::$G3;
177  $z1 = $z0 - $k1 + self::$G3;
178  $x2 = $x0 - $i2 + 2.0 * self::$G3; // Offsets for third corner in (x,y,z) coords
179  $y2 = $y0 - $j2 + 2.0 * self::$G3;
180  $z2 = $z0 - $k2 + 2.0 * self::$G3;
181  $x3 = $x0 - 1.0 + 3.0 * self::$G3; // Offsets for last corner in (x,y,z) coords
182  $y3 = $y0 - 1.0 + 3.0 * self::$G3;
183  $z3 = $z0 - 1.0 + 3.0 * self::$G3;
184 
185  // Work out the hashed gradient indices of the four simplex corners
186  $ii = $i & 255;
187  $jj = $j & 255;
188  $kk = $k & 255;
189  $gi0 = $this->perm[$ii + $this->perm[$jj + $this->perm[$kk]]] % 12;
190  $gi1 = $this->perm[$ii + $i1 + $this->perm[$jj + $j1 + $this->perm[$kk + $k1]]] % 12;
191  $gi2 = $this->perm[$ii + $i2 + $this->perm[$jj + $j2 + $this->perm[$kk + $k2]]] % 12;
192  $gi3 = $this->perm[$ii + 1 + $this->perm[$jj + 1 + $this->perm[$kk + 1]]] % 12;
193 
194  // Calculate the contribution from the four corners
195  $t0 = 0.6 - $x0 * $x0 - $y0 * $y0 - $z0 * $z0;
196  if($t0 < 0){
197  $n0 = 0.0;
198  }else{
199  $t0 *= $t0;
200  $n0 = $t0 * $t0 * self::dot3D(self::$grad3[$gi0], $x0, $y0, $z0);
201  }
202 
203  $t1 = 0.6 - $x1 * $x1 - $y1 * $y1 - $z1 * $z1;
204  if($t1 < 0){
205  $n1 = 0.0;
206  }else{
207  $t1 *= $t1;
208  $n1 = $t1 * $t1 * self::dot3D(self::$grad3[$gi1], $x1, $y1, $z1);
209  }
210 
211  $t2 = 0.6 - $x2 * $x2 - $y2 * $y2 - $z2 * $z2;
212  if($t2 < 0){
213  $n2 = 0.0;
214  }else{
215  $t2 *= $t2;
216  $n2 = $t2 * $t2 * self::dot3D(self::$grad3[$gi2], $x2, $y2, $z2);
217  }
218 
219  $t3 = 0.6 - $x3 * $x3 - $y3 * $y3 - $z3 * $z3;
220  if($t3 < 0){
221  $n3 = 0.0;
222  }else{
223  $t3 *= $t3;
224  $n3 = $t3 * $t3 * self::dot3D(self::$grad3[$gi3], $x3, $y3, $z3);
225  }
226 
227  // Add contributions from each corner to get the noise value.
228  // The result is scaled to stay just inside [-1,1]
229  return 32.0 * ($n0 + $n1 + $n2 + $n3);
230  }

232  public function getNoise2D($x, $y){
233  $x += $this->offsetX;
234  $y += $this->offsetY;
235 
236  // Skew the input space to determine which simplex cell we're in
237  $s = ($x + $y) * self::$F2; // Hairy factor for 2D
238  $i = (int) floor($x + $s);
239  $j = (int) floor($y + $s);
240  $t = ($i + $j) * self::$G2;
241  $X0 = $i - $t; // Unskew the cell origin back to (x,y) space
242  $Y0 = $j - $t;
243  $x0 = $x - $X0; // The x,y distances from the cell origin
244  $y0 = $y - $Y0;
245 
246  // For the 2D case, the simplex shape is an equilateral triangle.
247 
248  // Determine which simplex we are in.
249  if($x0 > $y0){
250  $i1 = 1;
251  $j1 = 0;
252  } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
253  else{
254  $i1 = 0;
255  $j1 = 1;
256  }
257  // upper triangle, YX order: (0,0)->(0,1)->(1,1)
258 
259  // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
260  // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y), where
261  // c = (3-sqrt(3))/6
262 
263  $x1 = $x0 - $i1 + self::$G2; // Offsets for middle corner in (x,y) unskewed coords
264  $y1 = $y0 - $j1 + self::$G2;
265  $x2 = $x0 + self::$G22; // Offsets for last corner in (x,y) unskewed coords
266  $y2 = $y0 + self::$G22;
267 
268  // Work out the hashed gradient indices of the three simplex corners
269  $ii = $i & 255;
270  $jj = $j & 255;
271  $gi0 = $this->perm[$ii + $this->perm[$jj]] % 12;
272  $gi1 = $this->perm[$ii + $i1 + $this->perm[$jj + $j1]] % 12;
273  $gi2 = $this->perm[$ii + 1 + $this->perm[$jj + 1]] % 12;
274 
275  // Calculate the contribution from the three corners
276  $t0 = 0.5 - $x0 * $x0 - $y0 * $y0;
277  if($t0 < 0){
278  $n0 = 0.0;
279  }else{
280  $t0 *= $t0;
281  $n0 = $t0 * $t0 * self::dot2D(self::$grad3[$gi0], $x0, $y0); // (x,y) of grad3 used for 2D gradient
282  }
283 
284  $t1 = 0.5 - $x1 * $x1 - $y1 * $y1;
285  if($t1 < 0){
286  $n1 = 0.0;
287  }else{
288  $t1 *= $t1;
289  $n1 = $t1 * $t1 * self::dot2D(self::$grad3[$gi1], $x1, $y1);
290  }
291 
292  $t2 = 0.5 - $x2 * $x2 - $y2 * $y2;
293  if($t2 < 0){
294  $n2 = 0.0;
295  }else{
296  $t2 *= $t2;
297  $n2 = $t2 * $t2 * self::dot2D(self::$grad3[$gi2], $x2, $y2);
298  }
299 
300  // Add contributions from each corner to get the noise value.
301  // The result is scaled to return values in the interval [-1,1].
302  return 70.0 * ($n0 + $n1 + $n2);
303  }
