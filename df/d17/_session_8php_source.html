103  public function update($time){
104  if(!$this->isActive and ($this->lastUpdate + 10) < $time){
105  $this->disconnect("timeout");
106  return;
107  }else{
108  $this->lastUpdate = $time;
109  }
110  $this->isActive = false;
111 
112  if(count($this->ACKQueue) > 0){
113  $pk = new ACK();

116  $this->ACKQueue = [];

161  protected function addToQueue(EncapsulatedPacket $pk, $flags = RakLib::PRIORITY_NORMAL){
162  $priority = $flags & 0b0000111;
163  if($pk->needACK and $pk->messageIndex !== null){
164  $this->needACK[$pk->identifierACK][$pk->messageIndex] = $pk->messageIndex;
165  }
166  if($priority === RakLib::PRIORITY_IMMEDIATE){ //Skip queues
167  $packet = new DATA_PACKET_0();
168  $packet->seqNumber = $this->sendSeqNumber++;
169  $packet->packets[] = $pk;
170  $this->sendPacket($packet);
171  $this->recoveryQueue[$packet->seqNumber] = $packet;
172  return;
173  }
174  $length = $this->sendQueue->length();
175  if($length + $pk->getTotalLength() > $this->mtuSize){

185  public function addEncapsulatedToQueue(EncapsulatedPacket $packet, $flags = RakLib::PRIORITY_NORMAL){
186 
187  $packet->needACK = ($flags & RakLib::FLAG_NEED_ACK) > 0;
188  $this->needACK[$packet->identifierACK] = [];
189 
190  if($packet->getTotalLength() + 4 > $this->mtuSize){
191  $packet->hasSplit = true;
192  $buffers = str_split($packet->buffer, $this->mtuSize - 34);
193  $packet->buffer = "";
194  $packet->splitID = $packet->splitIndex = ++$this->splitID % 65536;
195  $packet->splitCount = count($buffers);
196  $packet->reliability = 2;
197  foreach($buffers as $count => $buffer){
198  $pk = clone $packet;

256  public function handlePacket(Packet $packet){
257  $this->isActive = true;
258  if($this->state === self::STATE_CONNECTED or $this->state === self::STATE_CONNECTING_2){
259  if($packet::$ID >= 0x80 and $packet::$ID <= 0x8f and $packet instanceof DataPacket){ //Data packet
260  $packet->decode();
261  $diff = $packet->seqNumber - $this->lastSeqNumber;
262 
263  if($diff > static::$WINDOW_SIZE){
264  return;
265  }elseif($diff > 1){
266  for($i = $this->lastSeqNumber + 1; $i < $packet->seqNumber; ++$i){
267  $this->NACKQueue[$i] = $i;
268  }
269  }
270 
271  if($packet->seqNumber > $this->lastSeqNumber){
272  $this->lastSeqNumber = $packet->seqNumber;
273  }
274  unset($this->NACKQueue[$packet->seqNumber]);
275  $this->ACKQueue[$packet->seqNumber] = $packet->seqNumber;
276 
277  foreach($packet->packets as $pk){
278  $this->handleEncapsulatedPacket($pk);
279  }
280  }else{
281  if($packet instanceof ACK){
282  $packet->decode();
283  foreach($packet->packets as $seq){
284  if(isset($this->recoveryQueue[$seq])){
285  foreach($this->recoveryQueue[$seq]->packets as $pk){
286  if($pk->needACK and $pk->messageIndex !== null){
287  unset($this->needACK[$pk->identifierACK][$pk->messageIndex]);
288  }
289  }
290  unset($this->recoveryQueue[$seq]);
291  }
292  }
293  }elseif($packet instanceof NACK){
294  $packet->decode();
295  foreach($packet->packets as $seq){
296  if(isset($this->recoveryQueue[$seq])){
297  $this->sendPacket($this->recoveryQueue[$seq]);
298  }
299  }
300  }
301  }
302 
303  }elseif($packet::$ID > 0x00 and $packet::$ID < 0x80){ //Not Data packet :)
304  $packet->decode();
305  if($packet instanceof UNCONNECTED_PING){
306  $pk = new UNCONNECTED_PONG();
307  $pk->serverID = $this->sessionManager->getID();
308  $pk->pingID = $packet->pingID;
309  $pk->serverName = $this->sessionManager->getName();
310  $this->sendPacket($pk);
311  }elseif($packet instanceof OPEN_CONNECTION_REQUEST_1){
312  $packet->protocol; //TODO: check protocol number and refuse connections
313  $pk = new OPEN_CONNECTION_REPLY_1;
314  $pk->mtuSize = $packet->mtuSize;
315  $pk->serverID = $this->sessionManager->getID();
316  $this->sendPacket($pk);
317  $this->state = self::STATE_CONNECTING_1;
318  }elseif($this->state === self::STATE_CONNECTING_1 and $packet instanceof OPEN_CONNECTION_REQUEST_2){
319  $this->id = $packet->clientID;
320  if($packet->serverPort === $this->sessionManager->getPort()){
321  $this->mtuSize = min($packet->mtuSize, 1464); //Max size, do not allow creating large buffers to fill server memory
322  $pk = new OPEN_CONNECTION_REPLY_2;
323  $pk->mtuSize = $this->mtuSize;
324  $pk->serverID = $this->sessionManager->getID();
325  $pk->clientPort = $this->port;
326  $this->sendPacket($pk);
327  $this->state = self::STATE_CONNECTING_2;
328  }
329  }
330  }
